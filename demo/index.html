/* start: 调用slider幻灯片插件 */
//浮现样式
$("#slider_01").fadeSlider({
    controlsShow: true, //是否显示数字导航
    effect: 1, //1表示浮现,2表示滑动
    speed: 500, //滑动速度
    auto: true, //是否自定滑动
    pause: 3000, //两次滑动暂停时间持
    height: 0, //容器高度，不设置自动获取图片高度
    width: 0//容器宽度，不设置自动获取图片宽度
});
/* end */
